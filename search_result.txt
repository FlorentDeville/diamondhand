        original_quantity = el.data('quantity')
        new_quantity = el.val();

      if (original_quantity === new_quantity) {
        return;
      }

      if (!el.data('has_no_quantity_limit') && new_quantity > 4) {

        el.val(original_quantity);
        $('#cart-quantity-alert').show();
        $('#nav-quantity-alert').show();
        setTimeout(function() {
          $('#nav-quantity-alert').hide();
        }, 1500);
        return;

        url: el.data('href'),

            el.val(original_quantity);

        if (existing_item.length) {

          quantity_input.trigger('change');

          el.data('processing', false);

            success: function(response) {
